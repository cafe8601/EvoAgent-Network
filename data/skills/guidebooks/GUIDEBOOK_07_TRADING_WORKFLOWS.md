from statsmodels.tsa.stattools import coint

def find_cointegrated_pairs(prices_df, p_value_threshold=0.05):
    pairs = []
    tickers = prices_df.columns

    for i in range(len(tickers)):
        for j in range(i+1, len(tickers)):
            score, pvalue, _ = coint(prices_df[tickers[i]],
                                      prices_df[tickers[j]])
            if pvalue < p_value_threshold:
                pairs.append((tickers[i], tickers[j], pvalue))

    return sorted(pairs, key=lambda x: x[2])

def calculate_triangular_arbitrage(btc_usdt, eth_usdt, eth_btc, fee=0.001):
    # Path 1: USDT β†’ BTC β†’ ETH β†’ USDT
    usdt_to_btc = 1 / btc_usdt['ask'] * (1 - fee)
    btc_to_eth = usdt_to_btc / eth_btc['ask'] * (1 - fee)
    eth_to_usdt = btc_to_eth * eth_usdt['bid'] * (1 - fee)
    profit_1 = (eth_to_usdt - 1) * 100  # %

    # Path 2: USDT β†’ ETH β†’ BTC β†’ USDT
    usdt_to_eth = 1 / eth_usdt['ask'] * (1 - fee)
    eth_to_btc = usdt_to_eth * eth_btc['bid'] * (1 - fee)
    btc_to_usdt = eth_to_btc * btc_usdt['bid'] * (1 - fee)
    profit_2 = (btc_to_usdt - 1) * 100  # %

    return max(profit_1, profit_2)

class RiskManager:
    def __init__(self, config):
        self.max_daily_loss = config['max_daily_loss_pct']  # 0.05
        self.max_position_pct = config['max_position_pct']  # 0.20
        self.max_drawdown = config['max_drawdown_pct']      # 0.15
        self.min_risk_reward = config['min_risk_reward']    # 2.0

    def pre_trade_check(self, trade, portfolio):
        checks = {
            'daily_loss': portfolio.daily_pnl > -self.max_daily_loss,
            'position_size': trade.size_pct <= self.max_position_pct,
            'risk_reward': trade.risk_reward >= self.min_risk_reward,
            'stop_loss': trade.stop_loss is not None
        }

        return all(checks.values()), checks

    def portfolio_check(self, portfolio):
        if portfolio.drawdown > self.max_drawdown:
            return {'action': 'LIQUIDATE_ALL', 'reason': 'Max drawdown exceeded'}
        return {'action': 'CONTINUE'}
